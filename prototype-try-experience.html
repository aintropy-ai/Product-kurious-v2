💻 SDK / API

# Install the Kurious SDK pip install kurious from kurious import Client client = Client(api_key="sk-your-api-key-here") # Search across your data results = client.search( query="When did the witness change their testimony?", project="legal-video", mode="deep_dive", # or "quick" top_k=10 ) for r in results: print(f"{r.score:.2f} | {r.source} | {r.text[:100]}") # Find similar data (ML feature) similar = client.find_similar( reference="trajectory_1001", dimensions=["velocity", "force_torque"], top_k=20 ) # Edge case mining edge_cases = client.mine_edge_cases( project="robot-sim", threshold=0.3 )
// Install: npm install @aintropy/kurious import { Kurious } from '@aintropy/kurious'; const client = new Kurious({ apiKey: "sk-your-api-key-here" }); // Search your data const results = await client.search({ query: "When did the witness change their testimony?", project: "legal-video", mode: "deep_dive", topK: 10 }); results.forEach(r => { console.log(`${r.score} | ${r.source} | ${r.text}`); });
# REST API - Search endpoint curl -X POST https://api.aintropy.ai/v1/search \ -H "Authorization: Bearer sk-your-api-key-here" \ -H "Content-Type: application/json" \ -d '{ "query": "When did the witness change their testimony?", "project": "legal-video", "mode": "deep_dive", "top_k": 10 }' # Response includes: score, source, text, video_timestamp, confidence
# Batch API - Process multiple queries offline from kurious import BatchClient batch = BatchClient(api_key="sk-your-api-key-here") # Submit batch job (locally hosted option available) job = batch.submit( queries=["query1", "query2", ...], project="legal-video", output_format="jsonl", # Standard I/O, no platform lock-in max_budget=50.00 # Budget API: max spend limit ) # Check status status = batch.status(job.id) # Download when complete results = batch.download(job.id)
